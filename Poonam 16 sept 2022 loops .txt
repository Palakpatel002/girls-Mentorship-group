For loop


Que 1) write a program to print a number from 1 to 10.

#include <iostream>
using namespace std;

int main()
{
   for (int i=0; i<=10;i++)
   {
       cout<<i<<endl;
   }
   
    return 0;
}




Que 2) Write a program to print a table of input number.

#include <iostream>
using namespace std;

int main()
{ 
    cout<<"Enter any number";
    int n;
    cin>>n;
   for (int i=1; i<=10;i++)
   {
       cout<<n<<"*"<<i<<"="<<i*n<<endl;
   }
   
    return 0;
}

While loop


Que 5) write a program to print a sequence of numbers .

#include <iostream>
using namespace std;

int main()
{ 
    int i;
   while (i<=10)
   {
       cout<<i<<endl;
       i++;
   }
   
    return 0;
}
