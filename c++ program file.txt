//WRITE A PROGRAM TONREVERSE A NUMBER WITH WHILE LOOP:

#include <iostream>  
using namespace std;  
int main()  
{  
int n, c=0, rem;    
cout<<"Enter a number: ";    
cin>>n;    
  while(n!=0)    
  {    
     rem=n%10;      
     c=c*10+rem;    
     n/=10;    
  }    
 cout<<"Reversed Number: "<<c<<endl;     
return 0;  
}
